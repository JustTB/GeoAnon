public class SigmoidFunction extends ContinuousFunctionPlotter { 
 
    public double getY(double x) { 
 
        return 1 / (1 + Math.exp(-x)); 
 
    } 
 
    public String getName() { 
        return "Sigmoid Function"; 
    } 
 
 
}
